# How to create new rdv table?
1. Execute 'master_generate.sql" / stored procedure "master_generate_rdv"
	DECLARE	@return_value int
	EXEC	@return_value = [dbo].[master_generate_rdv] @execute_sql = 0
	SELECT	'Return Value' = @return_value
	GO


# How to drop an existing table(s)?
1. Execute sp "master_dropper_rdv"
	DECLARE	@return_value int
	EXEC	@return_value = [dbo].[master_dropper_rdv] @execute_sql = 0
	SELECT	'Return Value' = @return_value
	GO
